country_slot_weight = {
	if = {
		limit = {
			global_var:mp_limits_players_only = 1
			is_ai = yes
		}
		value = 0
	}
	else_if = {
		limit = {
			has_variable = mp_is_gp
		}
		value = global_var:mp_gp_cost_rule
	}
	else_if = {
		limit = {
			has_variable = mp_is_major
		}
		value = global_var:mp_major_cost_rule
	}
	else_if = {
		limit = {
			has_variable = mp_is_minor
		}
		value = global_var:mp_minor_cost_rule
	}
	else_if = {
		limit = {
			has_variable = mp_is_small
		}
		value = global_var:mp_small_cost_rule
	}
	else_if = {
		limit = {
			has_variable = mp_is_normal
		}
		value = global_var:mp_normal_cost_rule
	}
	else = { # Fallback
		value = 3
	}
}

total_player_alliance_weight = {
	value = 0
	# Current Alliance weight (exclude countries already in defensive league or PU with root, they count under DP)
	every_country = {
		limit = {
			has_mutual_scripted_relation = {
				type = relation_type:alliance
				target = root
			}
			# Skip if already in same defensive league
			not = {
				any_international_organizations_member_of = {
					international_organization_type = international_organization_type:defensive_league
					any_international_organization_member = {
						this = root
					}
				}
			}
			# Skip if in same personal union
			not = {
				any_international_organizations_member_of = {
					international_organization_type = international_organization_type:union
					any_international_organization_member = {
						this = root
					}
				}
			}
		}
		add = country_slot_weight
	}
	# Potential alliance weight
	if = {
		limit = {
			exists = scope:recipient
			not = { this = scope:recipient }
			not = {
				has_mutual_scripted_relation = {
					type = relation_type:alliance
					target = scope:recipient
				}
			}
		}
		add = scope:recipient.country_slot_weight
	}

	# Overflow from Defensive Pact weight
	# AP = Alliance Weight + max(0, Defensive Pact Weight - DP Limit)
	if = {
		limit = {
			total_player_defensive_pact_weight > mp_defensive_points_limit
		}
		add = {
			value = total_player_defensive_pact_weight
			subtract = mp_defensive_points_limit
		}
	}
}

		}

	}
