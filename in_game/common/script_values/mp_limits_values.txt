country_slot_weight = {
	# if = {
	# limit = { is_ai = yes }
	# value = 0
	# }
	if = {
		limit = {
			has_variable = mp_is_gp
		}
		value = 5
	}
	else_if = {
		limit = {
			has_variable = mp_is_major
		}
		value = 4
	}
	else = {
		value = 3
	}
}

mp_alliance_points_limit = {
	value = 6
}

mp_defensive_points_limit = {
	if = {
		limit = { has_variable = mp_is_gp }
		value = 1
	}
	else_if = {
		limit = { has_variable = mp_is_major }
		value = 2
	}
	else = {
		value = 3
	}
}

mp_guarantee_points_limit = {
	if = {
		limit = { has_variable = mp_is_gp }
		value = 1
	}
	else_if = {
		limit = { has_variable = mp_is_major }
		value = 2
	}
	else = {
		value = 3
	}
}

total_player_alliance_weight = {
	value = 0
	# Alliance weight
	every_country = {
		limit = {
			has_mutual_scripted_relation = {
				type = relation_type:alliance
				target = root
			}
		}
		add = country_slot_weight
	}
	# Potential alliance weight
	if = {
		limit = {
			exists = scope:recipient
			not = {
				has_mutual_scripted_relation = {
					type = relation_type:alliance
					target = scope:recipient
				}
			}
			# Check if this is an alliance relation being checked
			# We don't want to add potential weight if we are checking a defensive pact or guarantee
			# In scripted_relations/alliance.txt, can_add_alliance_rule is called
		}
		add = scope:recipient.country_slot_weight
	}

	# Overflow from Defensive Pact weight
	# AP = Alliance Weight + max(0, Defensive Pact Weight - DP Limit)
	if = {
		limit = {
			total_player_defensive_pact_weight > mp_defensive_points_limit
		}
		add = {
			value = total_player_defensive_pact_weight
			subtract = mp_defensive_points_limit
		}
	}
}

total_player_defensive_pact_weight = {
	value = 0
	# Count any relation that should use defensive points
	# For now, let's assume 'anti_coalition_treaty' or a custom one
	# If 'defensive_pact' doesn't exist, this will stay 0
	# every_country = {
	# 	limit = {
	# 		has_mutual_scripted_relation = {
	# 			type = relation_type:defensive_pact
	# 			target = root
	# 		}
	# 	}
	# 	add = country_slot_weight
	# }

	# Defensive League members
	every_country = {
		limit = {
			any_international_organizations_member_of = {
				international_organization_type = international_organization_type:defensive_league
				any_international_organization_member = {
					this = root
				}
			}
			not = { this = root }
		}
		add = country_slot_weight
	}
}

displayed_defensive_pact_weight = {
	value = total_player_defensive_pact_weight
	if = {
		limit = {
			total_player_defensive_pact_weight > mp_defensive_points_limit
		}
		value = mp_defensive_points_limit
	}
}

total_player_guarantee_weight = {
	value = 0
	every_country = {
		limit = {
			giving_scripted_relation = {
				type = relation_type:guarantee
				target = root
			}
		}
		add = country_slot_weight
	}
	# Potential guarantee weight
	if = {
		limit = {
			exists = scope:recipient
			not = {
				giving_scripted_relation = {
					type = relation_type:guarantee
					target = scope:recipient
				}
			}
		}
		add = scope:recipient.country_slot_weight
	}
}

is_over_alliance_limit = {
	value = 0
	if = {
		limit = {
			total_player_alliance_weight > mp_alliance_points_limit
		}
		value = 1
	}
}

is_over_defensive_limit = {
	value = 0
	if = {
		limit = {
			total_player_defensive_pact_weight > mp_defensive_points_limit
			total_player_alliance_weight >= mp_alliance_points_limit
		}
		value = 1
	}
}

is_over_guarantee_limit = {
	value = 0
	if = {
		limit = {
			total_player_guarantee_weight > mp_guarantee_points_limit
		}
		value = 1
	}
}
