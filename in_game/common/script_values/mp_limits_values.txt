# Mapping:
# Great Power (score > 10) = 3
# Major (score 2-10) = 2
# Minor (score < 2) = 1
# AI weight = 0

country_power_score = {
	value = 0

    # check sengoku_reevaluate_strongest_daimyos for relative power

	add = {
		value = country_pop_score
	}

	add = {
		value = country_eco_score
	}

	add = {
		value = country_manpower_score
	}

	add = {
		value = country_military_score
	}
}

country_pop_score = {
	value = 0

	add = {
		# take into account expected_army_size_modifier because of age modifier
		# value = expected_army_size
		# divide = 10000
		value = total_accepted_culture_population
		divide = 1000
	}
}

country_eco_score = {
	value = 0

	add = {
		value = country_tax_base
		divide = 500
	}
}

country_manpower_score = {
	value = 0

	add = {
		# value =
		value = max_manpower
		divide = 25
	}
}

country_military_score = {
	value = 0

	add = {
		value = regular_army_size
		divide = 10
	}
}

country_slot_weight = {
	if = {
		limit = {
			has_variable = mp_is_gp
		}
		value = 5
	}
	else_if = {
		limit = {
			has_variable = mp_is_major
		}
		value = 4
	}
	else = {
		value = 3
	}
}

mp_alliance_points_limit = {
	value = 6
}

mp_defensive_points_limit = {
	if = {
		limit = { has_variable = mp_is_gp }
		value = 1
	}
	else_if = {
		limit = { has_variable = mp_is_major }
		value = 2
	}
	else = {
		value = 3
	}
}

total_player_alliance_weight = {
	value = 0
	every_country = {
		limit = {
			giving_scripted_relation = {
				type = relation_type:alliance
				target = root
			}
			# is_ai = no
		}
		add = country_slot_weight
	}
	# Add the potential new ally's weight if this is called from a trigger checking a potential action
	if = {
		limit = {
			exists = scope:recipient
			not = {
				giving_scripted_relation = {
					type = relation_type:alliance
					target = scope:recipient
				}
			}
		}
		add = scope:recipient.country_slot_weight
	}
}

total_player_defensive_weight = {
	value = 0
	every_country = {
		limit = {
			giving_scripted_relation = {
				type = relation_type:guarantee
				target = root
			}
			# is_ai = no
		}
		add = country_slot_weight
	}
	# Add the potential new defensive partner's weight
	if = {
		limit = {
			exists = scope:recipient
			not = {
				giving_scripted_relation = {
					type = relation_type:guarantee
					target = scope:recipient
				}
			}
		}
		add = scope:recipient.country_slot_weight
	}
}

is_over_alliance_limit = {
	value = 0
	if = {
		limit = {
			total_player_alliance_weight > mp_alliance_points_limit
		}
		value = 1
	}
}

is_over_defensive_limit = {
	value = 0
	if = {
		limit = {
			total_player_defensive_weight > mp_defensive_points_limit
		}
		value = 1
	}
}
