mp_limits_toggle_tier_panel = {
	effect = {
		if = {
			limit = { has_global_variable = mp_tier_panel_open }
			remove_global_variable = mp_tier_panel_open
		}
		else = {
			set_global_variable = { name = mp_tier_panel_open value = 1 }
		}
	}
}

mp_limits_close_tier_panel = {
	effect = {
		remove_global_variable = mp_tier_panel_open
	}
}

# Per-country toggle for the "over alliance limit" popup warning. Operates on
# the calling country (root). Default state (variable absent) is muted.
mp_limits_toggle_alliance_alert = {
	effect = {
		root = {
			if = {
				limit = { has_variable = mp_alliance_alert_enabled }
				remove_variable = mp_alliance_alert_enabled
			}
			else = {
				set_variable = { name = mp_alliance_alert_enabled value = 1 }
			}
		}
	}
}
