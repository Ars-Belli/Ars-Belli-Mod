can_add_alliance_rule = {
	custom_description = {
		text = "MP_ALLIANCE_LIMIT_REACHED"
		object = scope:recipient
		OR = {
			is_ai = yes
			scope:recipient = { is_ai = yes }
			AND = {
				is_ai = no
				scope:recipient = { is_ai = no }
				total_player_alliance_weight <= 5
			}
		}
	}
}
