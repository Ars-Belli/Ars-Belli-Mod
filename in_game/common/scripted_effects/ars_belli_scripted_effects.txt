mp_limits_monthly_calculation = {
	# Check if it's a new month
	if = {
		limit = {
			OR = {
				not = { has_global_variable = mp_limits_last_month }
				not = { global_var:mp_limits_last_month = current_month }
			}
		}
		set_global_variable = {
			name = mp_limits_last_month
			value = current_month
		}
		remove_global_variable = mp_limits_calculation_done
	}

	# Only run once per month globally
	if = {
		limit = {
			not = {
				has_global_variable = mp_limits_calculation_done
			}
		}
		set_global_variable = mp_limits_calculation_done

		# Initialize rules globally
		mp_limits_rules_initialization = yes

		# Step 1: All countries calculate their basic scores
		set_global_variable = {
			name = mp_total_player_score
			value = 0
		}
		set_global_variable = {
			name = mp_player_count
			value = 0
		}

		every_country = {
			# Store other scores for UI
			set_variable = { name = mp_pop_score value = country_pop_score }
			set_variable = { name = mp_eco_score value = country_eco_score }
			set_variable = { name = mp_manpower_score value = country_manpower_score }
			set_variable = { name = mp_military_score value = country_military_score }

			# Calculate and store power score
			set_variable = {
				name = mp_power_score
				value = country_power_score
			}

			# Sum for average
			change_global_variable = {
				name = mp_total_player_score
				add = var:mp_power_score
			}
			change_global_variable = {
				name = mp_player_count
				add = 1
			}
		}

		# Step 2: Calculate average
		set_global_variable = {
			name = mp_avg_player_score
			value = global_var:mp_total_player_score
		}
		if = {
			limit = { global_var:mp_player_count > 0 }
			change_global_variable = {
				name = mp_avg_player_score
				divide = global_var:mp_player_count
			}
		}

		# Step 3: Calculate Ranks
		# Using ordered_in_global_list iterator for O(N log N) or O(N) internal sorting
		# instead of manual O(N^2) nested loops.
		clear_global_variable_list = mp_ranking_list
		every_country = {
			# Force initialization to prevent unset errors
			set_variable = { name = mp_rank_count value = 1000 }
			set_variable = { name = mp_rank_calculated_flag value = 0 } # Initialize to 0 explicitly
			add_to_global_variable_list = {
				name = mp_ranking_list
				target = this
			}
		}

		set_global_variable = {
			name = mp_rank_assigner
			value = 0
		}
		
		# Increment calculation counter
		if = {
			limit = { not = { has_global_variable = mp_limits_calc_counter } }
			set_global_variable = {
				name = mp_limits_calc_counter
				value = 0
			}
		}
		change_global_variable = {
			name = mp_limits_calc_counter
			add = 1
		}

		# Iterate and assign ranks
		# We need to manually loop through the list if ordered_in_global_list only returns one
		# But since we want to assign ranks to ALL, we use every_in_global_variable_list
		# and since it's NOT ordered, we have to do it differently.
		# Actually, ordered_in_global_list with a loop might work if we remove from list.
		
		# Alternative: use the O(N^2) approach again but optimized or use a different iterator
		# The user suggested ordered_country. Let's try that if it exists.
		# If it doesn't, we'll use the O(N^2) approach which is guaranteed to work.
		
		# Let's try O(N^2) again but simplified.
		every_country = {
			save_scope_as = target_country
			set_variable = { name = mp_rank_count value = 0 }
			every_country = {
				limit = {
					var:mp_power_score > scope:target_country.var:mp_power_score
				}
				scope:target_country = {
					change_variable = {
						name = mp_rank_count
						add = 1
					}
				}
			}
			set_variable = { name = mp_rank_calculated_flag value = 1 }
		}

		# Now classify every country based on their new rank
		every_country = {
			mp_limits_rank_classification = yes
		}
		
		clear_global_variable_list = mp_ranking_list
		remove_global_variable = mp_rank_assigner
	}
}

mp_limits_rank_classification = {
	# Local copy of global variables for easy access in UI/triggers if needed
	# Though they can also be accessed via global_var:
	set_variable = {
		name = mp_avg_player_score
		value = global_var:mp_avg_player_score
	}
	set_variable = {
		name = mp_gp_limit
		value = global_var:mp_gp_limit
	}
	set_variable = {
		name = mp_major_limit
		value = global_var:mp_major_limit
	}

	if = {
		limit = { NOT = { has_variable = mp_rank_count } }
		set_variable = { name = mp_rank_count value = 1000 }
	}
	if = {
		limit = { NOT = { has_variable = mp_rank_calculated_flag } }
		set_variable = { name = mp_rank_calculated_flag value = 0 }
	}

	set_variable = {
		name = mp_total_top_limit
		value = var:mp_gp_limit
	}
	change_variable = {
		name = mp_total_top_limit
		add = var:mp_major_limit
	}

	if = {
		limit = { var:mp_rank_count < var:mp_gp_limit }
		set_variable = { name = mp_is_gp value = yes }
		remove_variable = mp_is_major
		remove_variable = mp_is_small
		remove_variable = mp_is_minor
		remove_variable = mp_is_normal
	}
	else_if = {
		limit = { var:mp_rank_count < var:mp_total_top_limit }
		remove_variable = mp_is_gp
		set_variable = { name = mp_is_major value = yes }
		remove_variable = mp_is_small
		remove_variable = mp_is_minor
		remove_variable = mp_is_normal
	}
	else_if = {
		limit = {
			OR = {
				AND = {
					current_age = age_1_traditions
					total_population < 500
				}
				AND = {
					current_age = age_2_renaissance
					total_population < 600
				}
				AND = {
					current_age = age_3_discovery
					total_population < 700
				}
				AND = {
					current_age = age_4_reformation
					total_population < 800
				}
				AND = {
					current_age = age_5_absolutism
					total_population < 900
				}
				AND = {
					current_age = age_6_revolutions
					total_population < 1000
				}
			}
		}
		remove_variable = mp_is_gp
		remove_variable = mp_is_major
		remove_variable = mp_is_small
		set_variable = { name = mp_is_minor value = yes }
		remove_variable = mp_is_normal
	}
	else_if = {
		limit = {
			country_power_score < {
				value = var:mp_avg_player_score
				multiply = 0.5
			}
		}
		remove_variable = mp_is_gp
		remove_variable = mp_is_major
		set_variable = { name = mp_is_small value = yes }
		remove_variable = mp_is_minor
		remove_variable = mp_is_normal
	}
	else = {
		remove_variable = mp_is_gp
		remove_variable = mp_is_major
		remove_variable = mp_is_small
		remove_variable = mp_is_minor
		set_variable = { name = mp_is_normal value = yes }
	}

	if = {
		limit = {
			is_over_alliance_limit = 1
		}
		add_country_modifier = {
			modifier = mp_over_alliance_limit_penalty
		}
	}
	else = {
		remove_country_modifier = mp_over_alliance_limit_penalty
	}
}
