mp_limits_month_tick_reset = {
	# Check if it's a new month
	if = {
		limit = {
			OR = {
				not = { has_global_variable = mp_limits_last_month }
				not = { global_var:mp_limits_last_month = current_month }
			}
		}
		set_global_variable = {
			name = mp_limits_last_month
			value = current_month
		}
		remove_global_variable = mp_limits_calculation_done
	}
}

mp_limits_calculate_scores_and_avg = {
	# Step 1: All countries calculate their basic scores
	set_global_variable = {
		name = mp_total_player_score
		value = 0
	}
	set_global_variable = {
		name = mp_player_count
		value = 0
	}

	every_country = {
		limit = {
			OR = {
				global_var:mp_limits_players_only = 0
				is_ai = no
			}
		}
		# Store other scores for UI
		set_variable = { name = mp_pop_score value = country_pop_score }
		set_variable = { name = mp_eco_score value = country_eco_score }
		set_variable = { name = mp_trade_score value = country_trade_score }
		set_variable = { name = mp_manpower_score value = country_manpower_score }
		set_variable = { name = mp_sailors_score value = county_navy_score }
		set_variable = { name = mp_heavy_ships_score value = country_heavy_ship_score }
		set_variable = { name = mp_galleys_score value = country_galley_score }
		set_variable = { name = mp_control_pop_score value = country_control_pop_score }
		set_variable = { name = mp_vassal_pop_score value = country_vassal_pop_score }
		set_variable = { name = mp_military_score value = country_military_score }

		# Projected (instant) power score — what the smoothed score is tending toward.
		set_variable = {
			name = mp_power_score_projected
			value = country_power_score
		}

		# Smoothed displayed score: new = (old * 11 + projected) / 12, tends toward
		# projected over ~12 months. Snap to projected if unset or 0 (fresh game,
		# AI ↔ player flip, etc.) so it doesn't take a year to initialize.
		if = {
			limit = {
				OR = {
					NOT = { has_variable = mp_power_score }
					var:mp_power_score = 0
				}
			}
			set_variable = {
				name = mp_power_score
				value = var:mp_power_score_projected
			}
		}
		else = {
			change_variable = { name = mp_power_score multiply = 11 }
			change_variable = { name = mp_power_score add = var:mp_power_score_projected }
			change_variable = { name = mp_power_score divide = 12 }
		}

		# Sum for average
		change_global_variable = {
			name = mp_total_player_score
			add = var:mp_power_score
		}
		change_global_variable = {
			name = mp_player_count
			add = 1
		}
	}

	# Step 2: Calculate average
	set_global_variable = {
		name = mp_avg_player_score
		value = global_var:mp_total_player_score
	}
	if = {
		limit = { global_var:mp_player_count > 0 }
		change_global_variable = {
			name = mp_avg_player_score
			divide = global_var:mp_player_count
		}
	}
}

mp_limits_initialize_ranks_and_cleanup = {
	# Step 3: Initialize Ranks
	# If AI and players only mode, clear any potential markers/modifiers
	every_country = {
		if = {
			limit = {
				AND = {
					global_var:mp_limits_players_only = 1
					is_ai = yes
				}
			}
			remove_variable = mp_is_gp
			remove_variable = mp_is_major
			remove_variable = mp_is_small
			remove_variable = mp_is_minor
			remove_variable = mp_is_normal
			remove_variable = mp_rank_count
			remove_variable = mp_power_score
			remove_variable = mp_power_score_projected
			remove_variable = mp_pop_score
			remove_variable = mp_eco_score
			remove_variable = mp_trade_score
			remove_variable = mp_manpower_score
			remove_variable = mp_sailors_score
			remove_variable = mp_heavy_ships_score
			remove_variable = mp_galleys_score
			remove_variable = mp_control_pop_score
			remove_variable = mp_vassal_pop_score
			remove_variable = mp_military_score
			remove_country_modifier = mp_over_alliance_limit_penalty
		}
		else = {
			# Force initialization to prevent unset errors
			set_variable = { name = mp_rank_count value = 1000 }
		}
	}
}

mp_limits_assign_ranks = {
	# Step 4: Calculate and assign ranks
	# Using every_country with a filter and nested comparison for rank calculation.
	# This is O(N^2) but stable and correct in this environment.
	every_country = {
		save_scope_as = target_country
		limit = {
			OR = {
				global_var:mp_limits_players_only = 0
				is_ai = no
			}
		}
		set_variable = { name = mp_rank_count value = 0 }
		every_country = {
			limit = {
				OR = {
					global_var:mp_limits_players_only = 0
					is_ai = no
				}
				var:mp_power_score > scope:target_country.var:mp_power_score
			}
			scope:target_country = {
				change_variable = {
					name = mp_rank_count
					add = 1
				}
			}
		}
	}
}

mp_limits_game_start_effect = {
	# Initialize rules globally
	mp_limits_rules_initialization = yes

	# keep track of ai game rule, it shouldn't apply apply on game start (only 1337, not saves)
	if = {
		limit = {
			global_var:mp_limits_players_only = 1
		}
		set_variable = {
			name = mp_is_player_only_local
			value = 1
		}
		set_global_variable = { name = mp_limits_players_only value = 0 }
	}

	# on game start initialize all countries and ranks, including AI
	mp_limits_once_per_month_calculations = yes

	every_country = {
		mp_limits_rank_classification = yes
	}

	if = {
		limit = {
			var:mp_is_player_only_local = 1
		}
		set_variable = {
			name = mp_is_player_only_local
			value = 0
		}
		set_global_variable = { name = mp_limits_players_only value = 1 }
	}
}

mp_limits_once_per_month_calculations = {

	mp_limits_calculate_scores_and_avg = yes

	mp_limits_initialize_ranks_and_cleanup = yes

	mp_limits_assign_ranks = yes

	mp_limits_store_tier_lists = yes

	# Must run AFTER assign_ranks since country_slot_weight depends on rank vars.
	mp_limits_calculate_diplo_weights = yes
}

# Store countries per tier in score-descending lists. Iterated in GUI via
# `datamodel = "[GetGlobalList('mp_<tier>_list')]"` (see ars_belli_tier_list_panel).
# Two-pass: every_country has no order guarantee, so we collect into a temp list
# and re-iterate via ordered_in_list to get score-desc order.
mp_limits_store_tier_lists = {
	clear_global_variable_list = mp_gp_list
	clear_global_variable_list = mp_major_list
	clear_global_variable_list = mp_normal_list
	clear_global_variable_list = mp_small_list
	clear_global_variable_list = mp_minor_list
	clear_global_variable_list = mp_tier_sort_buffer

	every_country = {
		limit = {
			OR = {
				global_var:mp_limits_players_only = 0
				is_ai = no
			}
		}
		add_to_global_variable_list = { name = mp_tier_sort_buffer target = this }
	}

	ordered_in_global_list = {
		variable = mp_tier_sort_buffer
		order_by = var:mp_power_score
		max = 1000
		check_range_bounds = no

		if = {
			limit = { has_variable = mp_is_gp }
			add_to_global_variable_list = { name = mp_gp_list target = this }
		}
		else_if = {
			limit = { has_variable = mp_is_major }
			add_to_global_variable_list = { name = mp_major_list target = this }
		}
		else_if = {
			limit = { has_variable = mp_is_normal }
			add_to_global_variable_list = { name = mp_normal_list target = this }
		}
		else_if = {
			limit = { has_variable = mp_is_small }
			add_to_global_variable_list = { name = mp_small_list target = this }
		}
		else_if = {
			limit = { has_variable = mp_is_minor }
			add_to_global_variable_list = { name = mp_minor_list target = this }
		}
	}

	clear_global_variable_list = mp_tier_sort_buffer
}

mp_limits_assign_category_markers = {
	set_variable = {
		name = mp_total_top_limit
		value = var:mp_gp_limit
	}
	change_variable = {
		name = mp_total_top_limit
		add = var:mp_major_limit
	}

	if = {
		limit = { var:mp_rank_count < var:mp_gp_limit }
		set_variable = { name = mp_is_gp value = yes }
		remove_variable = mp_is_major
		remove_variable = mp_is_small
		remove_variable = mp_is_minor
		remove_variable = mp_is_normal
	}
	else_if = {
		limit = { var:mp_rank_count < var:mp_total_top_limit }
		remove_variable = mp_is_gp
		set_variable = { name = mp_is_major value = yes }
		remove_variable = mp_is_small
		remove_variable = mp_is_minor
		remove_variable = mp_is_normal
	}
	else_if = {
		limit = {
			OR = {
				AND = {
					current_age = age_1_traditions
					total_population < 350
				}
				AND = {
					current_age = age_2_renaissance
					total_population < 350
				}
				AND = {
					current_age = age_3_discovery
					total_population < 500
				}
				AND = {
					current_age = age_4_reformation
					total_population < 700
				}
				AND = {
					current_age = age_5_absolutism
					total_population < 1000
				}
				AND = {
					current_age = age_6_revolutions
					total_population < 1000
				}
			}
		}
		remove_variable = mp_is_gp
		remove_variable = mp_is_major
		remove_variable = mp_is_small
		set_variable = { name = mp_is_minor value = yes }
		remove_variable = mp_is_normal
	}
	else_if = {
		limit = {
			country_power_score < mp_small_power_threshold
		}
		remove_variable = mp_is_gp
		remove_variable = mp_is_major
		set_variable = { name = mp_is_small value = yes }
		remove_variable = mp_is_minor
		remove_variable = mp_is_normal
	}
	else = {
		remove_variable = mp_is_gp
		remove_variable = mp_is_major
		remove_variable = mp_is_small
		remove_variable = mp_is_minor
		set_variable = { name = mp_is_normal value = yes }
	}
}

mp_limits_apply_alliance_limit_penalty = {
	if = {
		limit = {
			global_var:mp_alliance_limit_penalty_enabled = 1
			is_over_alliance_limit = 1
			OR = {
				is_ai = no
				global_var:mp_limits_players_only = 0
			}
		}
		add_country_modifier = {
			modifier = mp_over_alliance_limit_penalty
		}

		# Pop up the warning event the first time the player crosses the limit,
		# but only if they've opted in via the tier list panel (default: off).
		# Re-arms when they drop back under.
		if = {
			limit = {
				is_human = yes
				NOT = { has_variable = mp_was_over_alliance_limit }
				has_variable = mp_alliance_alert_enabled
			}
			trigger_event_non_silently = ars_belli_mp_limits.1
		}
		set_variable = { name = mp_was_over_alliance_limit value = 1 }
	}
	else = {
		remove_country_modifier = mp_over_alliance_limit_penalty
		remove_variable = mp_was_over_alliance_limit
	}
}

mp_limits_rank_classification = {
	# If AI and players only mode, clear any potential markers/modifiers and exit
	if = {
		limit = {
			AND = {
				global_var:mp_limits_players_only = 1
				is_ai = yes
			}
		}
		remove_variable = mp_is_gp
		remove_variable = mp_is_major
		remove_variable = mp_is_small
		remove_variable = mp_is_minor
		remove_variable = mp_is_normal
		remove_variable = mp_rank_count
		remove_variable = mp_power_score
		remove_variable = mp_pop_score
		remove_variable = mp_eco_score
		remove_variable = mp_trade_score
		remove_variable = mp_manpower_score
		remove_variable = mp_sailors_score
		remove_variable = mp_heavy_ships_score
		remove_variable = mp_galleys_score
		remove_variable = mp_control_pop_score
		remove_variable = mp_vassal_pop_score
		remove_variable = mp_military_score
		remove_country_modifier = mp_over_alliance_limit_penalty
	}
	else = {
		# Local copy of global variables for easy access in UI/triggers if needed
		# Though they can also be accessed via global_var:
		set_variable = {
			name = mp_avg_player_score
			value = global_var:mp_avg_player_score
		}
		set_variable = {
			name = mp_gp_limit
			value = global_var:mp_gp_limit
		}
		set_variable = {
			name = mp_major_limit
			value = global_var:mp_major_limit
		}

		if = {
			limit = { NOT = { has_variable = mp_rank_count } }
			set_variable = { name = mp_rank_count value = 1000 }
		}

		mp_limits_assign_category_markers = yes
		mp_limits_apply_alliance_limit_penalty = yes
	}
}

# Precompute AP/DP/GP weights per country with proper dedup. Runs in effect context
# where `any_international_organizations_member_of` works (it errors with "Event target
# link 'international_organization_type' returned an invalid object" when used inside
# `every_country` in a script_value). Stores results in country variables which the
# `total_player_*` / `current_player_*` script_values just read back.
#
# Dedup rules (matching the mod's diplo-limit design):
#   - AP weight: alliance partners (each counted once, takes priority)
#   - DP weight: defensive-league + personal-union members NOT already counted in AP
#   - GP weight: countries this country is guaranteeing (no dedup)
mp_limits_calculate_diplo_weights = {
	every_country = {
		limit = {
			OR = {
				global_var:mp_limits_players_only = 0
				is_ai = no
			}
		}
		save_scope_as = mp_dw_p

		set_variable = { name = mp_ap_weight value = 0 }
		set_variable = { name = mp_dp_weight value = 0 }
		set_variable = { name = mp_gp_weight value = 0 }
		clear_variable_list = mp_dp_mates_buf
		clear_variable_list = mp_ap_partners
		clear_variable_list = mp_dp_partners
		clear_variable_list = mp_gp_partners

		# 1. Alliance partners (alliance > defensive league / PU for dedup)
		every_country = {
			limit = {
				has_mutual_scripted_relation = {
					type = relation_type:alliance
					target = scope:mp_dw_p
				}
				NOT = { this = scope:mp_dw_p }
			}
			save_temporary_scope_as = mp_dw_mate
			scope:mp_dw_p = {
				add_to_variable_list = {
					name = mp_dp_mates_buf
					target = scope:mp_dw_mate
				}
				add_to_variable_list = {
					name = mp_ap_partners
					target = scope:mp_dw_mate
				}
				change_variable = {
					name = mp_ap_weight
					add = { value = "scope:mp_dw_mate.country_slot_weight" }
				}
			}
		}

		# 2. Defensive League members (dedup against alliances)
		every_international_organizations_member_of = {
			limit = {
				international_organization_type = international_organization_type:defensive_league
			}
			every_international_organization_member = {
				limit = { NOT = { this = scope:mp_dw_p } }
				save_temporary_scope_as = mp_dw_mate
				scope:mp_dw_p = {
					if = {
						limit = {
							NOT = {
								is_target_in_variable_list = {
									name = mp_dp_mates_buf
									target = scope:mp_dw_mate
								}
							}
						}
						add_to_variable_list = {
							name = mp_dp_mates_buf
							target = scope:mp_dw_mate
						}
						add_to_variable_list = {
							name = mp_dp_partners
							target = scope:mp_dw_mate
						}
						change_variable = {
							name = mp_dp_weight
							add = { value = "scope:mp_dw_mate.country_slot_weight" }
						}
					}
				}
			}
		}

		# 3. Personal Union members (dedup against alliances + defensive league)
		every_international_organizations_member_of = {
			limit = {
				international_organization_type = international_organization_type:union
			}
			every_international_organization_member = {
				limit = { NOT = { this = scope:mp_dw_p } }
				save_temporary_scope_as = mp_dw_mate
				scope:mp_dw_p = {
					if = {
						limit = {
							NOT = {
								is_target_in_variable_list = {
									name = mp_dp_mates_buf
									target = scope:mp_dw_mate
								}
							}
						}
						add_to_variable_list = {
							name = mp_dp_mates_buf
							target = scope:mp_dw_mate
						}
						add_to_variable_list = {
							name = mp_dp_partners
							target = scope:mp_dw_mate
						}
						change_variable = {
							name = mp_dp_weight
							add = { value = "scope:mp_dw_mate.country_slot_weight" }
						}
					}
				}
			}
		}

		# 4. Guarantees (this country guarantees them)
		every_country = {
			limit = {
				receiving_scripted_relation = {
					type = relation_type:guarantee
					target = scope:mp_dw_p
				}
				NOT = { this = scope:mp_dw_p }
			}
			save_temporary_scope_as = mp_dw_mate
			scope:mp_dw_p = {
				add_to_variable_list = {
					name = mp_gp_partners
					target = scope:mp_dw_mate
				}
				change_variable = {
					name = mp_gp_weight
					add = { value = "scope:mp_dw_mate.country_slot_weight" }
				}
			}
		}

		clear_variable_list = mp_dp_mates_buf
	}
}
