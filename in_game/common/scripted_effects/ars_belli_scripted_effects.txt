mp_limits_monthly_calculation = {
	# Check if it's a new month
	if = {
		limit = {
			OR = {
				not = { has_global_variable = mp_limits_last_month }
				not = { global_var:mp_limits_last_month = current_month }
			}
		}
		set_global_variable = {
			name = mp_limits_last_month
			value = current_month
		}
		remove_global_variable = mp_limits_calculation_done
	}

	# Only run once per month globally
	if = {
		limit = {
			not = {
				has_global_variable = mp_limits_calculation_done
			}
		}
		set_global_variable = mp_limits_calculation_done

		# Initialize rules globally
		mp_limits_rules_initialization = yes

		# Step 1: All countries calculate their basic scores
		set_global_variable = {
			name = mp_total_player_score
			value = 0
		}
		set_global_variable = {
			name = mp_player_count
			value = 0
		}

		every_country = {
			# Store other scores for UI
			set_variable = { name = mp_pop_score value = country_pop_score }
			set_variable = { name = mp_eco_score value = country_eco_score }
			set_variable = { name = mp_manpower_score value = country_manpower_score }
			set_variable = { name = mp_military_score value = country_military_score }

			# Calculate and store power score
			set_variable = {
				name = mp_power_score
				value = country_power_score
			}

			# Sum for average
			change_global_variable = {
				name = mp_total_player_score
				add = var:mp_power_score
			}
			change_global_variable = {
				name = mp_player_count
				add = 1
			}

			# Increment calculation counter
			if = {
				limit = { not = { has_global_variable = mp_limits_calc_counter } }
				set_global_variable = {
					name = mp_limits_calc_counter
					value = 0
				}
			}
			change_global_variable = {
				name = mp_limits_calc_counter
				add = 1
			}
		}

		# Step 2: Calculate average
		set_global_variable = {
			name = mp_avg_player_score
			value = global_var:mp_total_player_score
		}
		if = {
			limit = { global_var:mp_player_count > 0 }
			change_global_variable = {
				name = mp_avg_player_score
				divide = global_var:mp_player_count
			}
		}

		# Step 3: Calculate Ranks
		# Using ordered_country iterator for O(N log N) or O(N) internal sorting
		# instead of manual O(N^2) nested loops.
		set_global_variable = {
			name = mp_rank_assigner
			value = 0
		}
		ordered_country = {
			order_by = {
				value = var:mp_power_score
			}
			# The iterator orders from highest value to lowest value by default.
			# We want rank 0 to be the highest score.
			set_variable = {
				name = mp_rank_count
				value = global_var:mp_rank_assigner
			}
			change_global_variable = {
				name = mp_rank_assigner
				add = 1
			}
		}
		remove_global_variable = mp_rank_assigner
	}
}
