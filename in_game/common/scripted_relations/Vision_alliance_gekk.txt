Vision_sharing = {
	type = diplomacy
	relation_type = mutual
    uses_diplo_capacity = no
    block_when_at_war = yes
    break_on_war = yes
    break_on_becoming_subject = yes
    lifts_fog_of_war = yes
    mutual_color = define:NMapColors|DIPLOMACY_ALLIANCE_COLOR
    category = {
        offer = CATEGORY_FRIENDLY_ACTIONS
        cancellation = CATEGORY_HOSTILE_ACTIONS
    }

    relation_type_for_ai = friendly
    show_break_alert = yes

	# Inlined alliance_can_exist minus the `opinion >= 0` checks on both sides —
	# vision sharing is intentionally available regardless of mutual opinion.
	offer_enabled = {
		scope:actor = {
			not = { has_blocked_treaties = scope:recipient }
			not = { is_subject_of = scope:recipient }
			not = { is_rival_of = scope:recipient }
			modifier:allowed_alliance = yes
			NOT = {
				is_member_of_international_organization_of_type = {
					type = coalition
					target = scope:recipient
				}
			}
			NOT = { in_union_with = scope:recipient }
			is_rebel_country = no
		}
		scope:recipient = {
			not = { has_blocked_treaties = scope:actor }
			not = { is_subject_of = scope:actor }
			not = { is_rival_of = scope:actor }
			modifier:allowed_alliance = yes
			NOT = {
				is_member_of_international_organization_of_type = {
					type = coalition
					target = scope:actor
				}
			}
			is_rebel_country = no
		}
		can_create_subject_alliance = {
			actor = scope:actor
			recipient = scope:recipient
		}
		NOT = {
			AND = {
				scope:actor = { is_hegemon = yes }
				scope:recipient = { is_hegemon = yes }
			}
		}
	}

	request_visible = { always = no }
	break_visible = { always = no }

    cancel_enabled = {
        scope:actor = {
            not = { is_fighting_war_together_with = scope:recipient }
        }
    }

    will_expire_trigger = {
    }

	offer_declined_effect = {
	}

    wants_to_give_diplo_chance = {
        actor_is_rival = -100
        actor_overlord_is_rival = -100
        base = -100
        #not_10_trust = -100
        trust_in_actor = -100
        positive_opinion = -100
        negative_opinion = -100
        border_distance = -100
        few_relations = -100
        same_court_language = -100
        royal_ties = -100
        strategic_interest = -100
        common_threat = -100
        relative_strength = -100
    }
}
