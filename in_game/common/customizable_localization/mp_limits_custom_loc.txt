mp_rank = {
	type = country
	text = {
		localization_key = MP_RANK_AI
		trigger = {
			is_ai = yes
			global_var:mp_limits_players_only = 1
		}
	}
	text = {
		localization_key = MP_RANK_GREAT
		trigger = { has_variable = mp_is_gp }
	}
	text = {
		localization_key = MP_RANK_MAJOR
		trigger = { has_variable = mp_is_major }
	}
	text = {
		localization_key = MP_RANK_MINOR_POWER
		trigger = { has_variable = mp_is_minor }
	}
	text = {
		localization_key = MP_RANK_SMALL
		trigger = { has_variable = mp_is_small }
	}
	text = {
		localization_key = MP_RANK_MINOR
		trigger = { has_variable = mp_is_normal }
	}
}

mp_rank_value = {
	type = country
	text = {
		localization_key = MP_RANK_VALUE_LOC
		trigger = {
			OR = {
				has_variable = mp_is_gp
				has_variable = mp_is_major
				has_variable = mp_is_minor
				has_variable = mp_is_small
				has_variable = mp_is_normal
			}
		}
	}
	text = {
		localization_key = MP_RANK_VALUE_LOC
		trigger = {
			NOT = { has_variable = mp_is_gp }
			NOT = { has_variable = mp_is_major }
			NOT = { has_variable = mp_is_minor }
			NOT = { has_variable = mp_is_small }
			NOT = { has_variable = mp_is_normal }
			OR = {
				is_ai = no
				global_var:mp_limits_players_only = 0
			}
		}
	}
}
