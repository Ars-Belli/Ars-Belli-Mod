monthly_country_pulse = {
	effect = {
		set_variable = {
			name = mp_pop_score
			value = country_pop_score
		}
		set_variable = {
			name = mp_eco_score
			value = country_eco_score
		}
		set_variable = {
			name = mp_manpower_score
			value = country_manpower_score
		}
		set_variable = {
			name = mp_military_score
			value = country_military_score
		}
		set_variable = {
			name = mp_power_score
			value = country_power_score
		}

		if = {
			# limit = { is_ai = no }
			set_variable = {
				name = mp_rank_count
				value = 0
			}
			# Count how many players have a higher power score
			every_country = {
				limit = {
					# is_ai = no
					country_power_score > root.country_power_score
				}
				root = {
					change_variable = {
						name = mp_rank_count
						add = 1
					}
				}
			}

			if = {
				limit = { var:mp_rank_count < 6 }
				set_variable = { name = mp_is_gp value = yes }
				remove_variable = mp_is_major
				remove_variable = mp_is_normal
			}
			else_if = {
				limit = { var:mp_rank_count < 12 }
				remove_variable = mp_is_gp
				set_variable = { name = mp_is_major value = yes }
				remove_variable = mp_is_normal
			}
			else = {
				remove_variable = mp_is_gp
				remove_variable = mp_is_major
				set_variable = { name = mp_is_normal value = yes }
			}
		}
	}
}
