monthly_country_pulse = {
	on_actions = {
		mp_limits_monthly_pulse
	}
}

mp_limits_monthly_pulse = {
	effect = {
		set_variable = {
			name = mp_pop_score
			value = country_pop_score
		}
		set_variable = {
			name = mp_eco_score
			value = country_eco_score
		}
		set_variable = {
			name = mp_manpower_score
			value = country_manpower_score
		}
		set_variable = {
			name = mp_military_score
			value = country_military_score
		}
		set_variable = {
			name = mp_power_score
			value = country_power_score
		}

		# set_variable = { # Debug
		# 	name = mp_debug_total_pop
		# 	value = total_population
		# }

		if = {
			# limit = { is_ai = no }
			set_variable = {
				name = mp_rank_count
				value = 0
			}
			set_variable = {
				name = mp_total_player_score
				value = 0
			}
			set_variable = {
				name = mp_player_count
				value = 0
			}

			# Count how many players have a higher power score and sum scores
			every_country = {
				limit = {
					# is_ai = no
				}
				root = {
					change_variable = {
						name = mp_total_player_score
						add = prev.country_power_score
					}
					change_variable = {
						name = mp_player_count
						add = 1
					}
					if = {
						limit = {
							prev.country_power_score > root.country_power_score
						}
						change_variable = {
							name = mp_rank_count
							add = 1
						}
					}
				}
			}

			set_variable = {
				name = mp_avg_player_score
				value = var:mp_total_player_score
			}
			if = {
				limit = { var:mp_player_count > 0 }
				change_variable = {
					name = mp_avg_player_score
					divide = var:mp_player_count
				}
			}

			if = {
				limit = { var:mp_rank_count < 6 }
				set_variable = { name = mp_is_gp value = yes }
				remove_variable = mp_is_major
				remove_variable = mp_is_small
				remove_variable = mp_is_minor
				remove_variable = mp_is_normal
			}
			else_if = {
				limit = { var:mp_rank_count < 12 }
				remove_variable = mp_is_gp
				set_variable = { name = mp_is_major value = yes }
				remove_variable = mp_is_small
				remove_variable = mp_is_minor
				remove_variable = mp_is_normal
			}
			else_if = {
				limit = {
					OR = {
						AND = {
							current_age = age_1_traditions
							total_population < 500
						}
						AND = {
							current_age = age_2_renaissance
							total_population < 600
						}
						AND = {
							current_age = age_3_discovery
							total_population < 700
						}
						AND = {
							current_age = age_4_reformation
							total_population < 800
						}
						AND = {
							current_age = age_5_absolutism
							total_population < 900
						}
						AND = {
							current_age = age_6_revolutions
							total_population < 1000
						}
					}
				}
				remove_variable = mp_is_gp
				remove_variable = mp_is_major
				remove_variable = mp_is_small
				set_variable = { name = mp_is_minor value = yes }
				remove_variable = mp_is_normal
			}
			else_if = {
				limit = {
					country_power_score < {
						value = var:mp_avg_player_score
						multiply = 0.5
					}
				}
				remove_variable = mp_is_gp
				remove_variable = mp_is_major
				set_variable = { name = mp_is_small value = yes }
				remove_variable = mp_is_minor
				remove_variable = mp_is_normal
			}
			else = {
				remove_variable = mp_is_gp
				remove_variable = mp_is_major
				remove_variable = mp_is_small
				remove_variable = mp_is_minor
				set_variable = { name = mp_is_normal value = yes }
			}
		}
	}
}
