force_break_union = {
	type = union
	category = CATEGORY_UNION_ACTIONS

	block_when_at_war = no

	potential = {
		scope:actor = {
			exists = union
		}
	}

	allow = {
		scope:actor = {
			has_ruler = yes
			custom_tooltip = {
				text = force_break_union_not_senior_tt
				OR = {
					union ?= {
						has_active_senior_partner = no
					}
					union ?= {
						has_active_senior_partner = yes
						NOT = { country_is_senior_partner = { country = scope:actor } }
					}
				}
			}
			custom_tooltip = {
				text = force_break_union_legitimacy_tt
				legitimacy >= 20
			}
		}
	}

	cooldown = {
		type = force_break_union_cd
		years = 50
	}

	select_trigger = {
		looking_for_a = country
		interaction_source_list = {
			scope:actor = {
				union ?= {
					every_international_organization_member = {
						limit = { NOT = { this = scope:actor } }
						add_to_list = source
					}
				}
			}
		}
		target_flag = recipient
		name = "choose_country"
		column = {
			data = name
		}
		visible = {
			in_union_with = scope:actor
		}
	}

	effect = {
		scope:actor = {
			add_legitimacy = legitimacy_extreme_penalty

			union ?= {
				remove_country_from_international_organization = scope:actor
			}

			create_character = {
				culture = scope:actor.culture
				religion = scope:actor.religion
				save_scope_as = ars_belli_new_random_ruler
			}

			set_new_ruler = scope:ars_belli_new_random_ruler
		}
	}

	ai_will_do = {
		add = -1000
	}
}
