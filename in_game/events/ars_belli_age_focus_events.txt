namespace = ars_belli_age_focus

# Second age-start focus selection — fires alongside vanilla ages_of_eu.1.
# Player picks 2 of 3 focuses across two chained events.
# Each pick sets a country flag (ars_belli_focus_expansion / _commerce / _warfare).
# Advances elsewhere gate on those flags directly (has_country_flag).
ars_belli_age_focus.1 = {
	type = age_event

	title = ars_belli_age_focus.1.title
	desc = ars_belli_age_focus.1.desc
	outcome = neutral

	illustration_tags = {
		10 = ages
	}

	immediate = {
		# Reset any prior age's picks so the player gets a clean 2-of-3 each age.
		clr_country_flag = ars_belli_focus_expansion
		clr_country_flag = ars_belli_focus_commerce
		clr_country_flag = ars_belli_focus_warfare
	}

	option = {
		name = ars_belli_age_focus.1.a   # Expansion
		set_country_flag = ars_belli_focus_expansion
		trigger_event_non_silently = { id = ars_belli_age_focus.2 }

		ai_will_select = { value = 100 }
	}

	option = {
		name = ars_belli_age_focus.1.b   # Commerce
		set_country_flag = ars_belli_focus_commerce
		trigger_event_non_silently = { id = ars_belli_age_focus.2 }

		ai_will_select = { value = 100 }
	}

	option = {
		name = ars_belli_age_focus.1.c   # Warfare
		set_country_flag = ars_belli_focus_warfare
		trigger_event_non_silently = { id = ars_belli_age_focus.2 }

		ai_will_select = { value = 100 }
	}
}

# Follow-up event — the second of the two picks. The option matching the
# already-picked focus is hidden via per-option trigger.
ars_belli_age_focus.2 = {
	type = age_event

	title = ars_belli_age_focus.2.title
	desc = ars_belli_age_focus.2.desc
	outcome = neutral

	illustration_tags = {
		10 = ages
	}

	option = {
		name = ars_belli_age_focus.1.a   # Expansion
		trigger = { NOT = { has_country_flag = ars_belli_focus_expansion } }
		set_country_flag = ars_belli_focus_expansion

		ai_will_select = { value = 100 }
	}

	option = {
		name = ars_belli_age_focus.1.b   # Commerce
		trigger = { NOT = { has_country_flag = ars_belli_focus_commerce } }
		set_country_flag = ars_belli_focus_commerce

		ai_will_select = { value = 100 }
	}

	option = {
		name = ars_belli_age_focus.1.c   # Warfare
		trigger = { NOT = { has_country_flag = ars_belli_focus_warfare } }
		set_country_flag = ars_belli_focus_warfare

		ai_will_select = { value = 100 }
	}
}
